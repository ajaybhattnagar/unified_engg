DELIMITER $$
CREATE TRIGGER ADD_PREMIUM_BALANCE
AFTER INSERT
   ON PARCELS FOR EACH ROW

BEGIN
	IF NEW.PREMIUM_AMOUNT > 0 THEN
	INSERT INTO `Sda`.`FEES`
	(`UNIQUE_ID`,
	`CATEGORY`,
	`AMOUNT`,
	`INTEREST`,
	`INTEREST_ACC_INTERVAL`,
	`EFFECTIVE_DATE`)
	VALUES
	(
	NEW.UNIQUE_ID,
	'2',
	NEW.PREMIUM_AMOUNT,
	NEW.PREMIUM_INTEREST,
	NEW.PREMIUM_INTERVAL,
	NEW.PREMIUM_EFFECTIVE_DATE);
	END IF;
END$$ 
DELIMITER ;
