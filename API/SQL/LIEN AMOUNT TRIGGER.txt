DELIMITER $$
CREATE TRIGGER ADD_BEGINNING_BALANCE
AFTER INSERT
   ON PARCELS FOR EACH ROW

BEGIN
	IF NEW.ORIGINAL_LIEN_AMOUNT > 0 THEN
	INSERT INTO `Sda`.`FEES`
	(`UNIQUE_ID`,
	`CATEGORY`,
	`AMOUNT`,
	`INTEREST`,
	`INTEREST_ACC_INTERVAL`,
	`EFFECTIVE_DATE`)
	VALUES
	(
	NEW.UNIQUE_ID,
	'1',
	NEW.ORIGINAL_LIEN_AMOUNT,
	NEW.ORIGINAL_LIEN_INTEREST,
	NEW.ORIGINAL_LIEN_INTERVAL,
	NEW.ORIGINAL_LIEN_EFFECTIVE_DATE);
	END IF;
END$$ 
DELIMITER ;

-- DROP TRIGGER Sda.ADD_BEGINNING_BALANCE;
